
# M2M_MP3_to_MIDI
Python script to turn MP3 into MIDI

# Usage

	python main.py input.mp3 -o output.mid
# Requirements

 - ffmpeg
 - pydub
 - numpy
 - scipy
 - mido
 - numba
# Installation

    git clone https://github.com/cylin577/M2M_MP3_to_MIDI
    cd M2M_MP3_to_MIDI
    pip install -r requirement.txt
